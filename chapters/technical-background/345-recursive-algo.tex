\begin{lstlisting}[caption=Recursive Connected Component Labelling (DFS), label=alg:ccl-recursive]
let img be the binary image
let lblimg be the labelled image

lblimg = negate(image)
lbl = 0

label_components():
  for i = 0 to height:
    for j = 0 to width:
      if lblimg[i, j] == -1:
        lbl += 1
        label\_region(i, j)

label_region(i, j):
  lblimg[i, j] = lbl
  for (i', j') in neighbours(i, j):
    if lblimg[i, j] == -1:
      label_region(i', j')
